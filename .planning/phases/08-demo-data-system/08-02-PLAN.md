// --- New state variables to add inside the Settings component ---
// Separate state for demo and clear confirm modals
const [showLoadConfirm, setShowLoadConfirm] = useState(false);
const [showClearConfirm, setShowClearConfirm] = useState(false);
const [demoSuccess, setDemoSuccess] = useState(false);

// --- Handler: confirm Load Demo Data ---
// Shows success banner, then reloads after 1500ms so banner is visible
function handleConfirmLoad() {
  setShowLoadConfirm(false);
  seedDemoData();
  setDemoSuccess(true);
  console.log('Demo data seeded — reloading in 1500ms');
  setTimeout(() => window.location.reload(), 1500);
}

// --- Handler: confirm Clear All Data ---
// Wipes data and immediately reloads (no success banner needed — page will show empty states)
function handleConfirmClear() {
  setShowClearConfirm(false);
  clearAllData();
  console.log('All data cleared — reloading');
  window.location.reload();
}
